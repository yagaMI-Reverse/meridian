MIGRATION — POLICIES & APPEND-ONLY TRIGGER

-- tenant id comes from the verified JWT, set per transaction
CREATE POLICY tenant_isolation ON ledger_entries
  USING (tenant_id = current_setting('app.tenant_id', true)::uuid);

-- unset GUC → NULL → matches nothing. Deny by default.

CREATE ROLE meridian_app LOGIN
  NOSUPERUSER NOBYPASSRLS;
GRANT SELECT, INSERT ON ledger_entries TO meridian_app;
-- note: no UPDATE, no DELETE. Ever.

CREATE TRIGGER ledger_entries_append_only
  BEFORE UPDATE OR DELETE ON ledger_entries
  FOR EACH ROW EXECUTE FUNCTION
    meridian_forbid_ledger_mutation();
